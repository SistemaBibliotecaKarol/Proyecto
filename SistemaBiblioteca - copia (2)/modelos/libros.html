function registrarLibro() { let titulo = document.getElementById("titulo").value; let autor = document.getElementById("autor").value; let categoria = document.getElementById("categoria").value; let editorial = document.getElementById("editorial").value; let disponible = document.getElementById("disponible").checked ? "Sí" : "No"; if(titulo=="" || autor=="" || categoria=="" || editorial==""){ alert("Complete todos los campos"); return; } let tabla = document.getElementById("tablaLibros").getElementsByTagName("tbody")[0]; let fila = tabla.insertRow(); fila.insertCell(0).innerHTML = titulo; fila.insertCell(1).innerHTML = autor; fila.insertCell(2).innerHTML = categoria; fila.insertCell(3).innerHTML = editorial; fila.insertCell(4).innerHTML = disponible; document.getElementById("titulo").value=""; document.getElementById("autor").value=""; document.getElementById("categoria").value=""; document.getElementById("editorial").value=""; document.getElementById("disponible").checked=false; alert("Libro registrado correctamente"); }
